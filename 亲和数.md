int main()
{//亲和数（互为完美数）
	int a,b,n,f=0 ;
	scanf("%d", &n);
	for (int j=1;j<=n;j++)
	{
		int s1=0,s2 = 0;
		scanf("%d%d", &a, &b);
		for (int i = 1; i < a; i++)
			if (a % i == 0)
				s1 += i;
		for (int i = 1; i < b; i++)
			if (b % i == 0)
				s2 += i;
		if (a == s2 && b == s1)
			printf("YES\n");
		else
			printf("NO\n");
	}
	return 0;
}
